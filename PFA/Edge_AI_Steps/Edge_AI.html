# Taille de la fentre en secondes (DOIT CORRESPONDRE  L'ESP32 ET AU DPLOIEMENT)
WINDOW_SIZE_SECONDS = 1
# Intervalle de lecture de l'ESP32 en ms (DOIT CORRESPONDRE  VOTRE CODE ESP32)
ESP32_READ_INTERVAL_MS = 100
# Nombre d'chantillons attendus par fentre
EXPECTED_SAMPLES_PER_WINDOW = int(WINDOW_SIZE_SECONDS * (1000 / ESP32_READ_INTERVAL_MS))

# Colonnes de donnes brutes  utiliser pour le CNN-LSTM
# Notez que nous utilisons l'amplitude du son et non une valeur brute
RAW_DATA_COLS = ['temperature_c', 'accel_x_g', 'accel_y_g', 'accel_z_g', 'sound_amplitude']


# --- 1. Charger et Pr-traiter les Donnes Brutes ---
print("Chargement des donnes brutes...")
df = pd.read_csv(RAW_DATA_FILE)

# Convertir la colonne de timestamp en format datetime et la dfinir comme index
df['timestamp_rpi'] = pd.to_datetime(df['timestamp_rpi'])
df = df.set_index('timestamp_rpi')

# Grer les valeurs sentinelles (-1.0 pour temprature)
df['temperature_c'] = df['temperature_c'].replace(-1.0, np.nan)
df = df.ffill().bfill() # Remplir les NaN avec la dernire/premire valeur
df = df.fillna(0) # Remplir les NaN restants (si au dbut de srie) avec 0

print("Donnes brutes charges et pr-traites.")
print(df.head())

# --- 2. Labellisation Manuelle (Base sur votre Journal de Bord) ---
# C'est la partie o vous devez diter le code en fonction de votre collecte !
print("\nDbut de la labellisation manuelle...")
df['fault_type'] = 'Normal' # Initialiser toutes les donnes comme 'Normal'

# Exemple de priodes d'anomalies ( ADAPTER AVEC VOS PROPRES TIMESTAMPS ET TYPES D'ANOMALIES)
# Utilisez les timestamps de votre journal de bord !
# Format: pd.to_datetime('AAAA-MM-JJ HH:MM:SS.ms')
anomalies = [
    {'start': '2025-08-07 14:55:10.371', 'end': '2025-08-07 15:00:05.085', 'type': 'Frottement'},
    {'start': '2025-08-07 16:14:14.704', 'end': '2025-08-07 16:20:07.254', 'type': 'Dsquilibre'},
    {'start': '2025-08-07 17:35:58.234', 'end': '2025-08-07 17:40:30.934', 'type': 'bruit'},

for anomaly in anomalies:
    start_ts = pd.to_datetime(anomaly['start'])
    end_ts = pd.to_datetime(anomaly['end'])
    df.loc[(df.index >= start_ts) & (df.index < end_ts), 'fault_type'] = anomaly['type']

print("\nLabellisation termine. Distribution des labels :")

# --- 3. Ingnierie des Caractristiques pour XGBoost ---
print("\nDbut de l'ingnierie des caractristiques pour XGBoost...")
features_list_xgb = []

for window_start, window_df in df.resample(f'{WINDOW_SIZE_SECONDS}s', origin='start_day', closed='right', label='right'):
    if not window_df.empty:
        window_label = window_df['fault_type'].mode()[0]

        # Assurez-vous que la fentre a suffisamment de donnes pour les calculs statistiques
        if len(window_df) < 2:
            pass

        features = {
            'label': window_label,
            'temp_mean': window_df['temperature_c'].mean(),
            'temp_std': window_df['temperature_c'].std(),
            'temp_min': window_df['temperature_c'].min(),
            'temp_max': window_df['temperature_c'].max(),

            # On utilise le RMS du son
            'sound_rms_mean': window_df['sound_amplitude'].mean(),
            'sound_rms_std': window_df['sound_amplitude'].std(),
            'sound_rms_max': window_df['sound_amplitude'].max(),

            'accel_x_rms': np.sqrt(np.mean(window_df['accel_x_g']**2)),
            'accel_y_rms': np.sqrt(np.mean(window_df['accel_y_g']**2)),
            'accel_z_rms': np.sqrt(np.mean(window_df['accel_z_g']**2)),
            
            'accel_x_std': window_df['accel_x_g'].std(),
            'accel_y_std': window_df['accel_y_g'].std(),
            'accel_z_std': window_df['accel_z_g'].std(),

            'accel_x_ptp': window_df['accel_x_g'].max() - window_df['accel_x_g'].min(),
            'accel_y_ptp': window_df['accel_y_g'].max() - window_df['accel_y_g'].min(),
            'accel_z_ptp': window_df['accel_z_g'].max() - window_df['accel_z_g'].min(),

            'accel_x_skew': skew(window_df['accel_x_g']),
            'accel_y_skew': skew(window_df['accel_y_g']),
            'accel_z_skew': skew(window_df['accel_z_g']),

            'accel_x_kurt': kurtosis(window_df['accel_x_g']),
            'accel_y_kurt': kurtosis(window_df['accel_y_g']),
            'accel_z_kurt': kurtosis(window_df['accel_z_g']),
        }
        features_list_xgb.append(features)

features_df_xgb = pd.DataFrame(features_list_xgb)
features_df_xgb = features_df_xgb.fillna(0.0) # Remplir les NaN rsultant des calculs de stats par 0

print("\nDataFrame des caractristiques pour XGBoost cr.")

xgb_feature_columns = features_df_xgb.drop('label', axis=1).columns.tolist()

X_dl = np.array(X_dl)
y_dl = np.array(y_dl)

print(f"Forme de y_dl (CNN-LSTM) : {y_dl.shape}")
print("Prparation des donnes brutes pour CNN-LSTM termine.")

    if len(window_data) == EXPECTED_SAMPLES_PER_WINDOW:

        X_dl.append(raw_features)
        y_dl.append(window_data['fault_type'].mode()[0])

X_dl = np.array(X_dl)
y_dl = np.array(y_dl)
y_dl_encoded = label_encoder.transform(y_dl)

scaler_dl = StandardScaler()
X_dl_reshaped = X_dl.reshape(-1, X_dl.shape[-1])
X_dl_scaled = scaler_dl.fit_transform(X_dl_reshaped)
X_dl_scaled = X_dl_scaled.reshape(X_dl.shape)

X_train_dl, X_test_dl, y_train_dl, y_test_dl = train_test_split(
    X_dl_scaled, y_dl_encoded, test_size=0.2, random_state=42, stratify=y_dl_encoded
)

print(f"Taille de l'ensemble d'entranement (X_train_dl) : {X_train_dl.shape}")
print(f"Taille de l'ensemble de test (X_test_dl) : {X_test_dl.shape}")

model_dl = Sequential([
    tf.keras.Input(shape=(X_train_dl.shape[1], X_train_dl.shape[2])),
    Conv1D(filters=64, kernel_size=3, activation='relu'),
    MaxPooling1D(pool_size=2),
    Dropout(0.2),

    LSTM(100, return_sequences=False),
    Dropout(0.3),

    Dense(len(label_encoder.classes_), activation='softmax')
])

model_dl.compile(optimizer='adam', loss='sparse_categorical_crossentropy', metrics=['accuracy'])

print("Entranement du modle CNN-LSTM...")
history_dl = model_dl.fit(X_train_dl, y_train_dl, epochs=50, batch_size=32, validation_split=0.1, verbose=1)
print("Entranement CNN-LSTM termin.")

loss_dl, accuracy_dl = model_dl.evaluate(X_test_dl, y_test_dl, verbose=0)
print(f"\n--- Rapports d'valuation CNN-LSTM ---")
print(f"Test Accuracy: {accuracy_dl:.4f}")

y_pred_dl = np.argmax(model_dl.predict(X_test_dl), axis=-1)
print("\nClassification Report (CNN-LSTM):\n", classification_report(y_test_dl, y_pred_dl, target_names=label_encoder.classes_))

cm_dl = confusion_matrix(y_test_dl, y_pred_dl)
plt.figure(figsize=(8, 6))
sns.heatmap(cm_dl, annot=True, fmt='d', cmap='Blues', xticklabels=label_encoder.classes_, yticklabels=label_encoder.classes_)
plt.title('Matrice de Confusion (CNN-LSTM)')
plt.xlabel('Prdit')
plt.ylabel('Rel')
plt.show()
